# Create H1 by using styled-component
const H1 = styled.h1`
    color: ${(props) => props.theme.textColor};
`;

# Set Dummpy function's type
interface DummyProps {
    text: string;

    # Make a optional props
    active?: boolean;
}

# active's default value is "false"
function Dummy({ text, active = false}) {
    return <H1>{text}</H1>
}

function App() {
    return (
        <Container>

            # this acitve mean active={true} in TS
            <Dummy active text="Hello!"/>
        </Container>
    )    
}

export default App;
